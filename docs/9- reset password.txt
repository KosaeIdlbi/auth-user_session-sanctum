من جدول ال password_reset_tokens الموجود مسبقا في لارافيل نعدله ليصبح 

Schema::create('password_reset_tokens', function (Blueprint $table) {
            $table->id();
            $table->string('email');
            $table->string('token');
            $table->timestamp('created_at')->nullable();
        });
